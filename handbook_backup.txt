	% ============================================
	\section{Introduction}
	% ============================================
	
	The \texttt{quickfig} package is a comprehensive LaTeX package designed to streamline academic writing, particularly in mathematics and technical fields. It provides:
	
	\begin{itemize}
		\item Extensive mathematical symbol libraries with consistent naming conventions
		\item Simplified figure and multi-figure insertion commands
		\item Pre-configured theorem environments
		\item Enhanced equation formatting tools
		\item Probability and statistical operators
		\item Color definitions for presentations
	\end{itemize}
	
	\subsection{Installation}
	
	Place \texttt{textool.sty} in your LaTeX project directory or in your local \texttt{texmf} tree. Then load it in your document preamble:
	
	\begin{lstlisting}
	\usepackage{textool}
	\end{lstlisting}
	
	\subsection{Package Philosophy}
	
	The package follows these design principles:
	\begin{enumerate}
		\item \textbf{Consistency}: All symbols follow predictable naming patterns
		\item \textbf{Simplicity}: Complex tasks require minimal syntax
		\item \textbf{Compatibility}: Works alongside other standard packages
		\item \textbf{Completeness}: Provides comprehensive symbol coverage
	\end{enumerate}

	QuickFig provides systematic access to mathematical symbols using consistent prefixes:

			\texttt{bbar} & Bar over symbol & \texttt{\textbackslash bbarx} $\rightarrow$ $\bbarx$ \\

	\subsection{Blackboard Bold Letters}
	
	Used primarily for number sets and spaces:
	
	\begin{example}
		\begin{lstlisting}
		$\mbR$ % Real numbers
		$\mbC$ % Complex numbers
		$\mbN$ % Natural numbers
		$\mbZ$ % Integers
		$\mbQ$ % Rational numbers
		$\mbP$ % Probability space
		\end{lstlisting}
	\end{example}
	
	Result: $\mbR$, $\mbC$, $\mbN$, $\mbZ$, $\mbQ$, $\mbP$

	\begin{example}

		$\ccalH$ % Hilbert space
		$\ccalL$ % Linear operators
		$\ccalM$ % Manifold
		\end{lstlisting}

	\section{Probability and Statistics}
	% ============================================
	
	\subsection{Expectation Operators}
	
	QuickFig provides multiple expectation notations:
	
	\begin{example}
		\begin{lstlisting}
		$\E{X}$          % E[X]
		$\Ec{X}$         % E(X) with parentheses
		$\Eb{X^2}$       % E with big brackets
		$\EB{X^2}$       % E with Big brackets
		$\EE{Y}{X|Y}$    % E_Y[X|Y] conditional expectation
		\end{lstlisting}
	\end{example}
	
	Result: $\E{X}$, $\Ec{X}$, $\Eb{X^2}$, $\EB{X^2}$, $\EE{Y}{X|Y}$
	
	\subsection{Probability Operators}
	
	\begin{example}
		\begin{lstlisting}
		$\Pr{X > 0}$     % P[X > 0]
		$\Prc{A \cup B}$ % P(A union B) with parentheses
		$\Pb{X = k}$     % P with big brackets
		$\PP{X}{Y|X}$    % P_X[Y|X] with subscript
		\end{lstlisting}
	\end{example}
	
	Result: $\Pr{X > 0}$, $\Prc{A \cup B}$, $\Pb{X = k}$, $\PP{X}{Y|X}$
	
	\subsection{Statistical Functions}
	
	\begin{example}
		\begin{lstlisting}
		$\Var{X}$              % Variance
		$\Covp{X}{Y}$          % Covariance
		$\ind{x > 0}$          % Indicator function
		$\Ind{A}$              % Alternative indicator
		$\Ent{X}$              % Entropy H[X]
		$\KL{P}{Q}$            % KL divergence
		$\MI{X}{Y}$            % Mutual information
		\end{lstlisting}
	\end{example}
	
	Result: $\Var{X}$, $\Covp{X}{Y}$, $\ind{x > 0}$, $\Ind{A}$

	\section{Figure Management}
	% ============================================
	
	\subsection{Single Figures}
	
	QuickFig simplifies figure insertion with brace-based syntax:
	
	\begin{example}
		\begin{lstlisting}
		% Basic figure (80% width)
		\fig{{image.png},{Caption text},{label}}
		
		% Custom width figure
		\figw[0.6]{{image.png},{Caption text},{label}}
		
		% Exact placement figure
		\figH{{image.png},{Caption text},{label}}
		
		% Figure without caption
		\fignc{{image.png},{label}}
		\end{lstlisting}
	\end{example}
	
	\begin{tip}
		Labels are automatically prefixed with \texttt{fig:}, so you reference them as \texttt{\textbackslash ref\{fig:label\}}
	\end{tip}
	
	\subsection{Multi-Figure Grids}
	
	QuickFig supports grid layouts from 1$\times$2 up to 4$\times$4:
	
	\subsubsection{2$\times$2 Grid Example}
	
	\begin{lstlisting}
	\mcfigTwoByTwo{{Overall caption},{main-label}}
	{{img1.png},{Caption 1},{label1}}
	{{img2.png},{Caption 2},{label2}}
	{{img3.png},{Caption 3},{label3}}
	{{img4.png},{Caption 4},{label4}}
	\end{lstlisting}

	\subsection{Spacing Control}
	
	Adjust spacing in multi-figure layouts:
	
	\begin{lstlisting}
	\setSubfigTopSkip{10pt}    % Space above figures
	\setSubfigBottomSkip{5pt}  % Space between rows
	\setSubfigCapSkip{3pt}     % Space below captions
	\end{lstlisting}

	\subsection{Available Environments}
	
	QuickFig provides pre-styled theorem environments:
	
	\begin{example}
		\begin{lstlisting}
		\begin{theorem}
		Every continuous function on a compact set is uniformly continuous.
		\end{theorem}
		
		\begin{lemma}
		If $f$ is differentiable, then $f$ is continuous.
		\end{lemma}
		
		\begin{definition}
		A set $S$ is \emph{compact} if every open cover has a finite subcover.
		\end{definition}
		\end{lstlisting}
	\end{example}

	\subsection{Custom Proof Environment}
	
	The \texttt{myproof} environment adds automatic QED symbols:
	
	\begin{example}
		\begin{lstlisting}
		\begin{myproof}
		Let $x \in S$. By assumption...
		Therefore, the statement holds.
		\end{myproof}
		
		\begin{myproof}[of Theorem 3.1]
		Using the previous lemma...
		\end{myproof}
		\end{lstlisting}
	\end{example}

	\subsection{Complete Mathematical Document}
	
	\begin{example}
		\begin{lstlisting}
		\documentclass{article}
		\usepackage{quickfig}
		
		\begin{document}
		
		\begin{theorem}
		Let $\bbX \in \mbR^{n \times m}$ be a random matrix with 
		$\E{\bbX} = \bbM$ and $\Var{\bbX_{ij}} = \sigma^2$.
		Then $\bbarX = \frac{1}{nm}\sum_{i,j} \bbX_{ij}$ 
		converges to $\tr(\bbM)/nm$ as $n,m \to \infty$.
		\end{theorem}
		
		\begin{myproof}
		By the law of large numbers, we have
		\feq{
		\Pr{\left|\bbarX - \E{\bbarX}\right| > \epsilon} 
		&\leq \frac{\Var{\bbarX}}{\epsilon^2} \\
		&= \frac{\sigma^2}{nm\epsilon^2} \to 0
		}
		as $n,m \to \infty$.
		\end{myproof}
		
		\end{document}
		\end{lstlisting}
	\end{example}

	\section{Troubleshooting}
	% ============================================
	
	\subsection{Common Issues}
	
	\begin{warning}
		\textbf{Algorithm Package Conflicts}: The modified version (1.1) checks for existing algorithm packages. If you need to use \texttt{algorithmic} instead of \texttt{algpseudocode}, load it before \texttt{quickfig}.
	\end{warning}
	
	\begin{warning}
		\textbf{Symbol Conflicts}: If a symbol is already defined by another package, quickfig will skip that definition. Load quickfig after other math packages if you prefer its definitions.
	\end{warning}
	
	\begin{warning}
		\textbf{UTF-8 Encoding}: If you encounter encoding errors, ensure you have \texttt{\textbackslash usepackage[utf8]\{inputenc\}} in your preamble.
	\end{warning}
	
	\subsection{Best Practices}
	
	\begin{enumerate}
		\item \textbf{Package Order}: Load \texttt{quickfig} after basic math packages but before document-specific definitions
		\item \textbf{Namespace}: The package uses protected namespaces for internal commands
		\item \textbf{Compatibility}: Version 1.1 includes compatibility checks for common packages
		\item \textbf{Encoding}: Always include proper encoding packages for international characters
	\end{enumerate}
